let count = 0; function moveNoButton() { count++; const noButton = document.getElementById('no-button'); const yesButton = document.getElementById('yes-button'); const buttonsDiv = document.getElementById('buttons'); const buttonsDivRect = buttonsDiv.getBoundingClientRect(); // Incrementar el tamaño del botón "Sí" yesButton.style.transform = `scale(calc(1 + 0.15 * ${count}))`; // Decrementar el tamaño del botón "No" noButton.style.transform = `scale(calc(1 - 0.05 * ${count}))`; // Evitar que el botón "No" se sobreponga al botón "Sí" const newLeft = Math.random() * (buttonsDivRect.width - noButton.offsetWidth); const newTop = Math.random() * (buttonsDivRect.height - noButton.offsetHeight); noButton.style.left = `${newLeft}px`; noButton.style.top = `${newTop}px`; }
